ECMAScript 2015?

Den meisten ist ECMAScript 2015 unter dem Namen ECMAScript 6 (kurz ES6) bekannt.

ECMA Standard

ECMAScript ist der Standard hinter JavaScript™

Jährliche Neuerung in Zukunft

In Zukunft wird der Standard jährlich aktualisiert und erweitert.

Stand 2015

Beinah alle Sprachfeatures herstellerübergreifend implementiert.

Stand 2016

Zusammenstellung Ende März 2016 abgeschlossen.

Breite Implementierung ist noch ausstehend.

Kategorisierung* der neuen Sprachfeatures

  • Syntax-Sugar
  • Bibliotheken
  • Spracherweiterungen

* Gegenerklärung: Meine persönliche Meinung und hat keinen Anspruch auf allgemeine Gültigkeit.

Syntax-Sugar

Default-Parameter

New way


							function foo(bar=42) { ... }

Old way


function foo(bar) {
	bar = (typeof bar === 'undefined') ? 42 : bar;
	...
}

Rest-Parameter

New way


function foo(bar, ...baz) { ... }

Old way


function foo(bar) {
	var baz = [], i;
	for(i=1; i < arguments.length; ++i) {
		baz.push(arguments[i]);
	}
	...
}

Spread-Operator

New way


var params = [42, 23];
function foo(bar, baz) { ... }
foo(...params);

Old way


var params = [42, 23];
function foo(bar, baz) { ... }
foo.apply(null, params);

Object-Schreibweisen

New way


var key = 'bar', baz = 42;
var obj = {
  foo() { ... },
  baz,
  [key]: 23
};

Old way


var key = 'bar', baz = 42;
var obj = {
  foo: function foo() { ... },
  baz: baz
};
obj[key] = 23;

Destructuring

New way


var obj = {foo: {bar: 23, baz: 42}};
var {foo: {bar, baz}} = obj;
var array = [1,2,3];
var [foo] = array;

Old way


var obj = {foo: {bar: 23, baz: 42}};
var bar = obj.foo.bar, baz = obj.foo.baz;
var array = [1,2,3];
var foo = array[0];

Template-String

New way


var bar = 23, baz = 42;
var foo = `${bar} + ${baz} = ${bar + baz}`;

Old way


var bar = 23, baz = 42;
var foo = '' + bar + ' + ' + baz + ' = ' + (bar + baz);

Arrow-Function

New way


var foo = [1,2,3,4,5,6]
  .filter((num) => num < 4)
  .map((num) => num * num);

Old way


var foo = [1,2,3,4,5,6]
  .filter(function(num) {return num < 4;})
  .map(function(num) {return num * num;});

Class

New way


class Foo {
  constructor(baz) {
    this.baz = baz
  }
  bar() { ... }
}

Old way


function Foo(baz) {
  this.baz = baz;
}
foo.prototype.bar = function bar() { ... };

Bibliotheken

Typed-Array


var int8 = new Int8Array(8);
var uint16 = new Uint16Array(4);
var float32 = new Float32Array(2);

Map, Set, WeakMap und WeakSet


var map = new Map();
map.set(2, 42);
map.set('2', 23);
map.size;
var set = new Set();
set.add(1);
set.add(1);
set.size;

Promise


asyncFunction(42)
  .then((result) => console.log(result))
  .catch((error) => console.error(error))
	.then(() => console.log('finally done'));

Spracherweiterungen

Blockscoping mit let und const


if (foo) {
  const bar = 42;
  let baz = 23;
  // ...
  baz = 21;
}
// bar und baz sind wieder undefined

Symbol


var sym1 = Symbol();
var sym2 = Symbol();
sym1 !== sym2;

Generator


function* gen() {
  var index = 0;
  while (true) {
    const input = yield index++;
    if(input) {
      break;
    }
  }
}
var generator = gen();
gen.next().value;
gen.next(true).done;

Proxy-Object


var handler = {
  get (target, key) {
    return key.charCodeAt(0);
  }
};
var target = {};
var charCodeOf = new Proxy(target, handler);
charCodeOf.a; // 97
charCodeOf.ä; // 228
charCodeOf['\u260E']; // ☎ 9742

ES Module


import _, {merge} from 'lodash';
import * as Foo from 'foo';

var api = {
  foo() { console.log('foo'); },
  bar(...args) {return merge({}, ...args); },
  lodash: _
};
export default api;
export function noop() {};
export function printTrace() {
  console.log('%s', new Error().stack);
};
