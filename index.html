Stage 4

Was kommt sicher in ECMAScript 2017.

Object.values/Object.entries


Object.keys({foo: 42, bar: 23}); // => ['foo', 'bar'], ES6/ES2015
Object.values({foo: 42, bar: 23}); // => [42, 23]
Object.entries({foo: 42, bar: 23}); // => [['foo', 42], ['bar', 23]]

Was könnte in ECMAScript 2017 sein.

SIMD (Single Instruction Multiple Data)

Parallelisierung von Rechenoperation in Vektoren


var a = SIMD.Float32x4(1.0, 2.0, 3.0, 4.0);
var b = SIMD.Float32x4(5.0, 10.0, 15.0, 20.0);
var c = SIMD.Float32x4.add(a,b); // => (6.0, 12.0, 18.0, 24.0)

Async/await

Syntax-Sugar für Promise-based Functions.


async function foo() {
  var result = await asyncFunc();
  return result;
}
foo().then((result) => console.log(result));

ES6/ES2015


function foo() {
  return Promise.resolve().then(() => {
    return asyncFunc();
  }).then((result) => {
    return result;
  });
}
foo().then((result) => console.log(result));

String.padStart/String.padEnd


'bar'.padStart(6, 'foo'); // => 'foobar'
'bar'.padStart(6); // => '   bar'
'foo'.padEnd(6, 'bar'); // => 'foobar'
'foo'.padEnd(6); // => 'foo   '
''.padStart(10, 'abc'); // => 'abcabcabca'

Async Iterators

Wie normale Iteratoren nur mit Promises und Syntax-Sugar.


async function* readLines(path) {
  let file = await fileOpen(path);
  try {
    while (!file.EOF)
      yield file.readLine();
  } finally {
    await file.close();
  }
}
for await (let line of readLines(path)) {
  console.log(line);
}
